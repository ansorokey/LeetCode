const b = "Robbie";
x = y = z = 0
[i, j] = [0, 0]

Printing:
```

```

Iteration:
```js
# Ranges
for( let i = 0; i < x; i++) {
    break;
}

for( let i = 0; i < a.length; i++) {
    break;
}

for( let i = start; i < stop; step) {
    break;
}

# Items
for( let key in thing) {
    break;
}
```
Conditionals
```js
if( a || b ){
    break;
}

if( a && b ){
    break;
}

if( !a ){
    break;
}
```
